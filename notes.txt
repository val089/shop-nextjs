## odpalenie kontenera z bazą danych w PostgreSQL
docker run --rm --name shop-nextjs-docker -e POSTGRES_PASSWORD=docker -d -p 5432:5432 -v $HOME/docker/volumes/postgres:/var/lib/postgresql/data postgres

## łączenie z bazą
psql -h localhost -U postgres -d postgres
